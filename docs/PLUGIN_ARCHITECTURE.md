    /// –†–µ–∑—É–ª—å—Ç–∞—Ç—ã –ø—Ä–æ–≤–µ—Ä–æ–∫

    /// TTL –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é (5 –º–∏–Ω—É—Ç)

struct CachedResult {
    result: AggregatedCheckResult,
    cached_at: Instant,
    ttl: Duration,
}

impl CheckCache {
    /// –ü–æ–ª—É—á–∏—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç –∏–∑ –∫—ç—à–∞ (–µ—Å–ª–∏ –Ω–µ –∏—Å—Ç—ë–∫)

        self.results.get(service_id).and_then(|cached| {
            if cached.cached_at.elapsed() < cached.ttl {
                Some(&cached.result)
            } else {
                None
            }
        })

    /// –°–æ—Ö—Ä–∞–Ω–∏—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç –≤ –∫—ç—à

        self.results.insert(service_id, CachedResult {
            result,
            cached_at: Instant::now(),
            ttl: self.default_ttl,
        });
    }

    /// –ò–Ω–≤–∞–ª–∏–¥–∏—Ä–æ–≤–∞—Ç—å –∫—ç—à –¥–ª—è —Å–µ—Ä–≤–∏—Å–∞

        self.results.remove(service_id);
    }

    /// –û—á–∏—Å—Ç–∏—Ç—å –≤–µ—Å—å –∫—ç—à
    pub fn clear(&mut self) {

    // ID –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å kebab-case
